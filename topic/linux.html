    /free  energy   TOTEN/  { print i++,$(NF-1); buffer = ""; next}
        {
        if (buffer != "") {
            print i++,buffer
            }
    buffer = $(NF-1)
            }
    END       {
        if (match($(NF-1), "^free  energy   TOTEN") == 0) {
            print i, buffer
            }
        }
